function [W1, W2, e] = MultiClass(W1, W2, X, D)
    alpha = 0.9;
    N = 5;
    for k = 1:N
        x = reshape(X(:, :, k), 40*40, 1);
        d = D(k, :)';
        v1 = W1*x;
        v1 = dlarray(v1);
        y1 = sigmoid(v1);
        v = W2*y1;
        v = extractdata(v);
        y = softmax(v);
        e = d - y;
        delta = e;
        e1 = W2'*delta;
        delta1 = y1.*(1-y1).*e1;
        dW1 = alpha*delta1*x';
        W1 = W1 + dW1;
        dW2 = alpha*delta*y1';
        W2 = W2 + dW2;
    end
end

function imagem = funcao_ler_imagem(name, numrow, numcol)
    img = imread(name);
    img =255-img;
    img = im2double(img);
    J = imresize(img, [numrow, numcol]);
    J = rgb2gray(J);
    J = imbinarize(J);
    imagem = J;

end
